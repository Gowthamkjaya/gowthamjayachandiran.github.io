function showSkillCategory(categoryId) {
  // Hide all skill content sections
  const skillContents = document.querySelectorAll('.skill-content');
  skillContents.forEach(content => {
    content.classList.remove('active');
  });
  
  // Show the selected skill content section
  document.getElementById(categoryId).classList.add('active');
  
  // Update button active state
  const buttons = document.querySelectorAll('.tab-button');
  buttons.forEach(button => {
    button.classList.remove('active');
  });
  
  // Add active class to clicked button
  event.currentTarget.classList.add('active');
}

// Expand/collapse timeline items
document.addEventListener('DOMContentLoaded', function() {
  const expandButton = document.getElementById('expand-timeline');
  
  expandButton.addEventListener('click', function() {
    const collapsedItems = document.querySelectorAll('.timeline-item.collapsed');
    
    collapsedItems.forEach(item => {
      item.classList.toggle('collapsed');
    });
    
    if (expandButton.innerHTML.includes('Show More')) {
      expandButton.innerHTML = 'Show Less <i class="fas fa-chevron-up"></i>';
    } else {
      expandButton.innerHTML = 'Show More Experience <i class="fas fa-chevron-down"></i>';
    }
  });
});

// Add scroll animation
document.addEventListener('DOMContentLoaded', function() {
  const sections = document.querySelectorAll('.section-header');
  
  const observerOptions = {
    root: null,
    rootMargin: '0px',
    threshold: 0.1
  };
  
  const observer = new IntersectionObserver(function(entries, observer) {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        const icon = entry.target.querySelector('.section-icon');
        icon.classList.add('animate__animated', 'animate__bounceIn');
        observer.unobserve(entry.target);
      }
    });
  }, observerOptions);
  
  sections.forEach(section => {
    observer.observe(section);
  });
});
</script>
